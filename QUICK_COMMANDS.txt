【一键打包】
./package_for_appstore.sh

【分步执行】
./build_framework.sh          # 构建 Framework
./verify_framework.sh         # 验证 Framework
./configure_signing.sh        # 配置签名

【查看文档】
cat APP_STORE_SUBMISSION_GUIDE.md    # App Store 提交指南
cat CHECKLIST.md                      # 检查清单
cat README.md                         # 项目说明
cat SUMMARY.md                        # 工作总结

【Flutter 命令】
flutter pub get               # 获取依赖
flutter clean                 # 清理构建
flutter run                   # 运行应用
flutter build ios             # 构建 iOS

【CocoaPods 命令】
cd ios && pod install         # 安装依赖
cd ios && pod update          # 更新依赖

【验证签名】
codesign --verify --deep --strict ./build/frameworks/Release/App.xcframework

【重新签名】
codesign --force --sign "Apple Distribution" --timestamp ./build/frameworks/Release/App.xcframework

【提交流程】
1. ./package_for_appstore.sh
2. 集成 frameworks 到 OC 项目
3. 配置 Build Settings 和签名
4. 真机测试
5. Product → Archive
6. Distribute App → App Store Connect
7. 提交审核
